Check if date is today and hide/show class

        var someDate = new Date("1 Nov 2017").toDateString();
        var today = new Date().toDateString();
        if (today === someDate) {
            jQuery('.late').fadeIn();
            jQuery('.early').fadeOut();
        } else if (today != someDate) {
            jQuery('.late').fadeOut();
            jQuery('.early').fadeIn();
        }
